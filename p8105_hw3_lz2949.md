``` r
sum_nhanes = m_nhanes |>
  rowwise() |>
  mutate(sum_acti = sum(c(min1:min1440))) |>
  select(-6:-1445)

ggplot(sum_nhanes, aes(x = age, y = sum_acti, color = sex)) +
  geom_point() +
  geom_smooth(method = "lm", formula = y ~ x, se = FALSE) +
```
